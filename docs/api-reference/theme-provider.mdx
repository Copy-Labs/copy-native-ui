# ThemeProvider

`ThemeProvider` is the root component that provides theme context to your application.

## Import

```tsx
import { ThemeProvider } from 'radix-native-ui';
```

## Usage

```tsx
import { ThemeProvider } from 'radix-native-ui';

function App() {
  return (
    <ThemeProvider>
      <AppContent />
    </ThemeProvider>
  );
}
```

## Props

### Theme Configuration

<PropTable>
  <Prop name="accentColor" type="Color" default="'indigo'">
    Primary accent color for the theme.
  </Prop>
  <Prop name="grayColor" type="'mauve' | 'olive' | 'sage' | 'sand' | 'slate' | 'gray'" default="'mauve'">
    Gray scale color.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'" default="'medium'">
    Border radius scale.
  </Prop>
  <Prop name="scaling" type="number" default="1">
    Overall scaling factor (1 = 100%).
  </Prop>
  <Prop name="panelBackground" type="'solid' | 'translucent'" default="'solid'">
    Background style for panels.
  </Prop>
</PropTable>

### Runtime Behavior

<PropTable>
  <Prop name="appearance" type="'light' | 'dark' | 'inherit'" default="'inherit'">
    Color scheme preference. Use 'inherit' to follow device/parent settings.
  </Prop>
  <Prop name="forcedMode" type="'light' | 'dark'">
    Force a specific mode, overriding device and parent settings.
  </Prop>
  <Prop name="hasBackground" type="boolean" default="true">
    Whether to apply background color.
  </Prop>
  <Prop name="onModeChange" type="(mode: 'light' | 'dark') => void">
    Callback when theme mode changes.
  </Prop>
</PropTable>

### Legacy Support

<PropTable>
  <Prop name="themeOptions" type="Partial<Theme>">
    Theme options object (legacy - use flat props instead). If provided alongside flat props, flat props take precedence.
  </Prop>
</PropTable>

### Toast Configuration

<PropTable>
  <Prop name="toastConfig" type="ToastConfig">
    Configuration options for toast notifications.
  </Prop>
</PropTable>

## Examples

### Basic Setup

```tsx
import { ThemeProvider, Text } from 'radix-native-ui';

function App() {
  return (
    <ThemeProvider>
      <Text>Hello World</Text>
    </ThemeProvider>
  );
}
```

### Custom Accent Color

```tsx
<ThemeProvider accentColor="blue">
  <App />
</ThemeProvider>
```

### Dark Mode

```tsx
<ThemeProvider appearance="dark">
  <App />
</ThemeProvider>
```

### System Appearance

```tsx
<ThemeProvider appearance="inherit">
  <App />
</ThemeProvider>
```

### Custom Radius

```tsx
<ThemeProvider radius="large">
  <App />
</ThemeProvider>
```

### Custom Gray Scale

```tsx
<ThemeProvider grayColor="gray">
  <App />
</ThemeProvider>
```

### Full Configuration

```tsx
<ThemeProvider
  accentColor="crimson"
  grayColor="mauve"
  panelBackground="translucent"
  radius="large"
  scaling="105%"
  appearance="dark"
>
  <App />
</ThemeProvider>
```

### Dynamic Theme

```tsx
function App() {
  const [appearance, setAppearance] = useState('light');
  const [accentColor, setAccentColor] = useState('blue');

  return (
    <ThemeProvider
      appearance={appearance}
      accentColor={accentColor}
    >
      <Flex direction="column" gap={3}>
        <SegmentedControl.Root
          value={appearance}
          onValueChange={setAppearance}
        >
          <SegmentedControl.Item value="light">Light</SegmentedControl.Item>
          <SegmentedControl.Item value="dark">Dark</SegmentedControl.Item>
        </SegmentedControl.Root>

        <Select value={accentColor} onValueChange={setAccentColor}>
          <Select.Item value="blue">Blue</Select.Item>
          <Select.Item value="green">Green</Select.Item>
          <Select.Item value="crimson">Crimson</Select.Item>
        </Select>

        <AppContent />
      </Flex>
    </ThemeProvider>
  );
}
```

### Nested Themes

```tsx
<ThemeProvider accentColor="blue">
  <Flex direction="column" gap={3}>
    <Text>Blue theme</Text>

    <ThemeProvider accentColor="green">
      <Text>Green theme (nested)</Text>
    </ThemeProvider>
  </Flex>
</ThemeProvider>
```
